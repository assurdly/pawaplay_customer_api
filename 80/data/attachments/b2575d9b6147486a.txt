//Response error messsage not properly catered for
const response = pm.response.json();

pm.test('Status code is 400', function () {
    pm.response.to.have.status(400);
});

pm.test('Response body contains valid error message', function () {
    pm.expect(response).to.have.property('errors');
    pm.expect(response.errors).to.be.an('object').that.is.not.empty;
    pm.expect(response.errors).to.have.property('message');
});
