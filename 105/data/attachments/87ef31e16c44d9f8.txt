const jsonData = pm.response.json();

pm.test("Response status code is 200", () => {
    pm.response.to.have.status(200);
});

pm.test("Response body contains expected success fields and message", () => {
    pm.expect(jsonData).to.have.property("data").that.is.a("string").and.is.not.empty;
    pm.expect(jsonData).to.have.property("message", "Token was reissued successfully.");
    pm.expect(jsonData).to.have.property("isSuccessful", true);
});

pm.test("Response structure is correct", () => {
    pm.expect(jsonData).to.have.property("messages").that.is.an("array");
    pm.expect(jsonData).to.have.property("timeGenerated").that.is.a("string");
});
