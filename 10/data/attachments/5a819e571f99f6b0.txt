const jsonData = pm.response.json();

pm.test("Response status code is 404", () => {
    pm.response.to.have.status(404);
});

pm.test("Response body contains expected fields", () => {
    pm.expect(jsonData).to.have.property("data", "");
    pm.expect(jsonData).to.have.property("isSuccessful", false);
});

pm.test("Response message indicates phone number is not connected", () => {
    pm.expect(jsonData.message).to.include("not connected to a profile");
});

pm.test("Response structure is correct", () => {
    pm.expect(jsonData.messages).to.be.an("array").and.be.empty;
    pm.expect(jsonData.timeGenerated).to.be.a("string");
});
